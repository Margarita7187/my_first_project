# Базовый сервер - тестовые запросы

# Главная страница
curl -X GET http://localhost:3000/

# Получить всех пользователей
curl -X GET http://localhost:3000/api/users

# Получить все продукты  
curl -X GET http://localhost:3000/api/products

# Получить конкретного пользователя
curl -X GET http://localhost:3000/api/user/1

# Отправить данные
curl -X POST http://localhost:3000/api/data \
  -H "Content-Type: application/json" \
  -d '{"name": "Test User", "email": "test@example.com"}'

# Несуществующий маршрут
curl -X GET http://localhost:3000/unknown

# Продвинутый сервер - тестовые запросы

# Главная страница
curl -X GET http://localhost:3001/

# Получить всех пользователей
curl -X GET http://localhost:3001/api/users

# Создать пользователя
curl -X POST http://localhost:3001/api/users \
  -H "Content-Type: application/json" \
  -d '{"name": "Новый пользователь", "email": "new@example.com", "role": "user"}'

# Обновить пользователя
curl -X PUT http://localhost:3001/api/users/1 \
  -H "Content-Type: application/json" \
  -d '{"name": "Обновленное имя", "email": "updated@example.com"}'

# Удалить пользователя
curl -X DELETE http://localhost:3001/api/users/1

# Получить логи
curl -X GET http://localhost:3001/api/logs
